
  CREATE SEQUENCE  SEQ_NM1  MINVALUE 1 MAXVALUE 999999 INCREMENT BY 1 START WITH 1 CACHE 2 NOORDER  NOCYCLE  NOPARTITION ;
  CREATE TABLE NOMINATION_ADDITION 
   (
	EVT_ID VARCHAR2(20 BYTE), 
	LBL_NAME VARCHAR2(200 BYTE), 
	LBL_TYPE VARCHAR2(500 BYTE), 
	DEF_VALUE VARCHAR2(1000 BYTE), 
	ID VARCHAR2(20 BYTE), 
	ISMANDATE VARCHAR2(20 BYTE)
   );


  CREATE OR REPLACE EDITIONABLE TRIGGER TRIG_NOMINATION 
   before insert on NOMINATION_ADDITION 
   for each row 
begin  
   if inserting then 
      if :NEW.ID is null then 
         select SEQ_NM1.nextval into :NEW.ID from dual; 
      end if; 
   end if; 
end;


ALTER TRIGGER TRIG_NOMINATION ENABLE;
